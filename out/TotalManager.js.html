records.TotalManager = function (me) {
    this.setMe = function (_me) {
        me = _me;
    };

    function getDiscount(item, category_total) {
        var cat_discount;
        Assert.isPositiveNum(item.discount, 'discount');
        switch (item.discount_type) {
        case "PERCENT":
            {
                cat_discount = category_total * item.discount / 100;
                break;
            }
        case "NUMBER":
            {
                cat_discount = item.discount;
            }
        }
        if (isNaN(cat_discount)) {
            cat_discount = 0;
        }
        return Utility.parseFloat(cat_discount);
    }

    function getTaxPercentage(name) {
        for (var i = 0; i < Settings.taxes.length; i++) {
            if (name == Settings.taxes[i].name) {
                return Utility.parseFloat(Settings.taxes[i].value);
            }
        };
        return 0;
    }

    function gettotalTax(taxes, onTotal) {
        if (!taxes) {
            return 0;
        }
        var perTax = 0,
            value = 0;
        for (var k = 0; k < taxes.length; k++) {
            value = taxes[k].tax_percent || pregetTaxPercentage(taxes[k].name);
            Assert.isPositiveNum(value, 'tax_percent');
            perTax += parseFloat(value);
        }
        return Utility.parseFloat(onTotal * perTax / 100);
    }

    /**
     *
     * @test {"args":[{"quantity": 2, "unit_cost": 100, "discount_type": "NUMBER", "discount": 10}, {"discount": 0,"total":0, "gtotal":0 }], "expected": 0 }
     */
    this.getTotal = function (data, obj) {
        var category_total = 0,
            cat_discount = 0;
        Assert.isPositiveNum(data.quantity, 'quantity');
        category_total = Assert.isPositiveNum(data.unit_cost || 0, 'unit_cost') * Assert.isGreaterThan(data.quantity, 0, 'quantity');
        cat_discount = getDiscount(data, category_total);
        Assert.isLessOrEqual(cat_discount, category_total, 'discount', "Discount is more than total.");
        obj.discount += cat_discount;
        obj.total += category_total;
        var totalAfterDiscount = (category_total - cat_discount);
        obj.tax += gettotalTax(data.taxes, totalAfterDiscount);
        obj.gtotal += totalAfterDiscount + gettotalTax(data.taxes, totalAfterDiscount);
    }
};
